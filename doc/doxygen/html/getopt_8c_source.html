51 /*
52  * getopt --
53  * Parse argc/argv argument vector.
54  */
55 int
56 BSDgetopt(int nargc, char *const *nargv, const char *ostr)
57 {
58  static char *place = EMSG; /* option letter processing */
59  char *oli; /* option letter list index */
60 
61  if (ostr == NULL)
62  return (-1);
63 
64  if (BSDoptreset || !*place) { /* update scanning pointer */
65  BSDoptreset = 0;
66  if (BSDoptind >= nargc || *(place = nargv[BSDoptind]) != '-') {
67  place = EMSG;
68  return (-1);
69  }
70  if (place[1] && *++place == '-') { /* found "--" */
71  if (place[1])
72  return (BADCH);
73  ++BSDoptind;
74  place = EMSG;
75  return (-1);
76  }
77  } /* option letter okay? */
78  if ((BSDoptopt = (int)*place++) == (int)':' ||
79  !(oli = strchr(ostr, BSDoptopt))) {
80  /*
81  * if the user didn't specify '-' as an option,
82  * assume it means -1.
83  */
84  if (BSDoptopt == (int)'-')
85  return (-1);
86  if (!*place)
87  ++BSDoptind;
88  if (BSDopterr && *ostr != ':')
89  (void)fprintf(stderr,
90  "%s: unknown option -- %c\n", __progname, BSDoptopt);
91  return (BADCH);
92  }
93  if (*++oli != ':') { /* don't need argument */
94  BSDoptarg = NULL;
95  if (!*place)
96  ++BSDoptind;
97  }
98  else { /* need an argument */
99  if (*place) /* no white space */
100  BSDoptarg = place;
101  else if (nargc <= ++BSDoptind) { /* no arg */
102  place = EMSG;
103  if (*ostr == ':')
104  return (BADARG);
105  if (BSDopterr)
106  (void)fprintf(stderr,
107  "%s: option requires an argument -- %c\n",
108  __progname, BSDoptopt);
109  return (BADCH);
110  }
111  else /* white space */
112  BSDoptarg = nargv[BSDoptind];
113  place = EMSG;
114  ++BSDoptind;
115  }
116  return (BSDoptopt); /* dump back option letter */
117 }
